\ \
F = \frac{ \left( Z - b \right) b / a_e + a_e }{ \sqrt{ X^2 + Y^2 } }, 
\\
&  P = \frac{4}{3} \left( E F + 1  \right), 
\ \
Q = 2 \left( E^2 - F^2 \right), 
\ \
D = P^3 + Q^2 .\end{align*}
